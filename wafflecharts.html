data =[{'Cuota de mercado': 66, '-': 34},
      {'Cuota de mercado': 12, '-': 88},
      {'Cuota de mercado': 11, '-': 89},
      {'Cuota de mercado': 8, '-': 92},
      {'Cuota de mercado': 3, '-': 97}]

navegadores = ['Chrome', 'Edge', 'Safari', 'Firefox', 'Ópera']
colores = ['#42f46e', '#42e2f4', '#4295f4', '#f4a742', '#f50a0a']

for i in range(0,5):
    
    # Creación del gráfico
    fig = plt.figure(
        figsize=(5,5),
        FigureClass=Waffle,  # Clase Waffle de pywaffle
        rows=10,
        values=data[i],
        colors=[colores[i], '#e3dcdc'],  # Colores por categoría
        legend={'loc': 'upper left', 'bbox_to_anchor': (1.1, 1)},  #Leyenda
        title={'label': 'Navegador: '+ navegadores[i], 'loc': 'left'},
        vertical=True
    )

    plt.text(0.45,0.45,str(data[i]['Cuota de mercado'])+'%', weight='bold', fontsize=18) #Texto en medio del gráfico

    plt.show()
